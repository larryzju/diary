for … of

for (let a of [1,2,3]) {
    console.log(a)
}

与 for ... in 不同，后者会获取 object 中所有 index

Object.prototype.name = function(){};
foo = [1,2,3];
foo.bar = "bar";

for (let k in foo) {
    console.log(k);
}

for (let k of foo) {
    console.log(k);
}
